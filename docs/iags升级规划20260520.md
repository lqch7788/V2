### 4.3 biz 文件编写模式

每个 V1.1 API 路由改为一个独立的 biz 文件，遵循 iAGS biz 模式：

```javascript
// 文件: D:\iAGS\tm.iags_biz\biz\iAGS\v11\crop\getVarietyList.{m}.js
const yjDBService = global.yjRequire("yujiang.Foil").yjDBService;
const yjDB = global.yjRequire("yujiang.Foil").yjDB;

module.exports = function(sender) {
    let query = sender.req.query;  // broker 传入的查询参数
    let page = parseInt(query.page) || 1;
    let pageSize = parseInt(query.pageSize) || 20;

    yjDBService.exec({
        sql: `SELECT * FROM V11_CropVarieties WHERE Status='active'
              LIMIT ? OFFSET ?`,
        parameters: [pageSize, (page - 1) * pageSize],
        success: function(result) {
            let data = yjDB.dataSet2ObjectList(result.meta, result.rows);
            sender.success({ rows: data, total: result.rows.length });
        },
        error: function(err) {
            console.log(err);
            sender.error(err);
        }
    });
};
```

### 4.4 改造优先级

```typescript
// src/services/brokerClient.ts — 新项目中的统一 API 客户端

const IAGS_BASE = '/app/system/broker'

interface BrokerResponse<T> {
  success?: boolean
  data?: T
  rows?: T[]
  total?: number
  [key: string]: any
}

/**
 * GET 请求（通过 broker getFromBiz）
 * @param params bizURLParams 三段式数组 ["iAGS","module","action"]
 * @param query URL 查询参数
 */
export async function brokerGet<T>(params: string[], query: Record<string, any> = {}): Promise<BrokerResponse<T>> {
  const sp = new URLSearchParams()
  sp.set('bizURLParams', JSON.stringify(params))
  // iAGS broker 的数据参数名是 data
  sp.set('data', JSON.stringify(query))

  const res = await fetch(`${IAGS_BASE}/getFromBiz?${sp.toString()}`, {
    credentials: 'include',  // 自动携带 iAGS Cookie
  })

  if (!res.ok) {
    throw new Error(`Broker GET failed: ${res.status} ${res.statusText}`)
  }

  return res.json()
}

/**
 * POST 请求（通过 broker post2Biz）
 * @param params bizURLParams 三段式数组
 * @param data POST body 数据
 */
export async function brokerPost<T>(params: string[], data: Record<string, any> = {}): Promise<BrokerResponse<T>> {
  const res = await fetch(`${IAGS_BASE}/post2Biz`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    credentials: 'include',
    body: JSON.stringify({
      bizURLParams: params,
      data: data,
    }),
  })

  if (!res.ok) {
    throw new Error(`Broker POST failed: ${res.status} ${res.statusText}`)
  }

  return res.json()
}

/**
 * PUT 请求
 */
export async function brokerPut<T>(params: string[], data: Record<string, any> = {}): Promise<BrokerResponse<T>> {
  const res = await fetch(`${IAGS_BASE}/post2Biz`, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    credentials: 'include',
    body: JSON.stringify({
      bizURLParams: params,
      data: data,
    }),
  })

  if (!res.ok) {
    throw new Error(`Broker PUT failed: ${res.status} ${res.statusText}`)
  }

  return res.json()
}

/**
 * DELETE 请求
 */
export async function brokerDelete<T>(params: string[], query: Record<string, any> = {}): Promise<BrokerResponse<T>> {
  const sp = new URLSearchParams()
  sp.set('bizURLParams', JSON.stringify(params))
  sp.set('data', JSON.stringify(query))

  const res = await fetch(`${IAGS_BASE}/getFromBiz?${sp.toString()}`, {
    method: 'DELETE',
    credentials: 'include',
  })
```

### 6.2 postMessage 协议定义

```typescript
// ===== 消息类型定义 =====

// Shell → iframe 消息
type ShellToIframeMessage =
  | ShellReadyMessage       // Shell 初始化完成
  | RouteChangeMessage      // 路由/菜单切换
  | ThemeChangeMessage      // 主题切换
  | UserInfoMessage         // 用户信息同步
  | BreadcrumbMessage       // 面包屑更新

// iframe → Shell 消息
type IframeToShellMessage =
  | IframeReadyMessage      // iframe 加载完成
  | IframeResizeMessage     // iframe 内容高度变化
  | TitleUpdateMessage      // 页面标题更新
  | NavigationMessage       // iframe 内导航请求
  | ErrorMessage            // iframe 加载错误

// ===== 消息格式 =====

interface IagsMessage<T = any> {
  source: 'iags-shell' | 'iags-legacy'  // 消息来源
  type: string                            // 消息类型
  payload: T                              // 消息数据
  timestamp: number                       // 时间戳
  messageId: string                       // 唯一消息ID（用于调试）
}

// ===== 具体消息类型 =====

// Shell → iframe: 用户信息
interface UserInfoPayload {
  userOID: string
  userName: string
  userAID: string
  orgOID: string
  orgName: string
  roles: string[]
  accessToken: string     // 传递给 iframe 用于 API 调用
}

// Shell → iframe: 路由变更
interface RouteChangePayload {
  module: string
  page: string
  params: Record<string, string>
}

// Shell → iframe: 主题变更
interface ThemeChangePayload {
  theme: 'light' | 'dark'
  primaryColor: string
  fontSize: number
}

// iframe → Shell: 标题更新
interface TitleUpdatePayload {
  title: string           // 页面标题
  breadcrumb: string[]    // 面包屑路径
}

// iframe → Shell: 高度变化
interface IframeResizePayload {
  height: number          // iframe 内容实际高度 (px)
}

// iframe → Shell: 错误
interface ErrorPayload {
  code: string
  message: string
  detail?: string
}
```

### 6.3 IframeContainer 组件实现

```typescript
// src/components/legacy/IframeContainer.tsx

import { useEffect, useRef, useState, useCallback } from 'react'
import { useAuthStore } from '@/stores/useAuthStore'

interface IframeContainerProps {
  src: string                    // iAGS 旧页面 URL
  moduleKey: string              // 模块标识（用于日志）
  onTitleChange?: (title: string, breadcrumb: string[]) => void
  onError?: (error: ErrorPayload) => void
}

export function IframeContainer({ src, moduleKey, onTitleChange, onError }: IframeContainerProps) {
  const iframeRef = useRef<HTMLIFrameElement>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const user = useAuthStore((s) => s.user)

  // 监听 iframe 的 postMessage
  useEffect(() => {
    const handleMessage = (event: MessageEvent) => {
      // 安全校验：只接受同源消息
      if (event.origin !== window.location.origin) return

      const msg = event.data as IagsMessage

      // 只接受来自 legacy 的消息
      if (msg.source !== 'iags-legacy') return

      switch (msg.type) {
        case 'IFRAME_READY':
          setLoading(false)
          setError(null)
          // iframe 就绪后立即发送用户信息
          sendToIframe({
            source: 'iags-shell',
            type: 'USER_INFO',
            payload: {
              userOID: user?.userOID,
              userName: user?.userName,
              userAID: user?.userAID,
              orgOID: user?.orgOID,
              orgName: user?.orgName,
              roles: user?.roles || [],
              accessToken: user?.accessToken || '',
            },
            timestamp: Date.now(),
            messageId: crypto.randomUUID(),
          })
          break

        case 'TITLE_UPDATE':
          onTitleChange?.(msg.payload.title, msg.payload.breadcrumb)
          break

        case 'IFRAME_RESIZE':
          // 动态调整 iframe 高度
          if (iframeRef.current) {
            iframeRef.current.style.height = `${msg.payload.height}px`
          }
          break

        case 'ERROR':
          setError(msg.payload.message)
          onError?.(msg.payload)
          break

        case 'NAVIGATION':
          // iframe 内请求导航到其他模块
          // 通过 React Router 处理
          break
      }
    }

    window.addEventListener('message', handleMessage)
    return () => window.removeEventListener('message', handleMessage)
  }, [user, onTitleChange, onError])

  // 发送消息给 iframe
  const sendToIframe = useCallback((message: IagsMessage) => {
    if (iframeRef.current?.contentWindow) {
      iframeRef.current.contentWindow.postMessage(message, window.location.origin)
    }
  }, [])

  // 处理加载错误
  const handleLoadError = () => {
    setLoading(false)
    setError(`模块 "${moduleKey}" 加载失败`)
    onError?.({ code: 'LOAD_ERROR', message: `模块加载失败: ${src}` })
  }

  return (
    <div className="iframe-container relative w-full h-full">
      {/* 加载态 */}
      {loading && (
        <div className="absolute inset-0 flex items-center justify-center bg-gray-50">
          <div className="text-center">
            <div className="animate-spin h-8 w-8 border-2 border-blue-500 border-t-transparent rounded-full mx-auto" />
            <p className="mt-2 text-sm text-gray-500">正在加载 {moduleKey} 模块...</p>
          </div>
        </div>
      )}

      {/* 错误态 */}
      {error && (
        <div className="absolute inset-0 flex items-center justify-center bg-gray-50">
          <div className="text-center">
            <p className="text-red-500 mb-2">{error}</p>
            <button
              className="px-4 py-2 bg-blue-500 text-white rounded hover:bg-blue-600"
              onClick={() => {
                setError(null)
                setLoading(true)
                // 重新加载 iframe
                if (iframeRef.current) {
                  iframeRef.current.src = src
                }
              }}
            >
              重新加载
            </button>
          </div>
        </div>
      )}

      {/* iframe */}
      <iframe
        ref={iframeRef}
        src={src}
        className="w-full border-none"
        style={{ minHeight: '600px' }}
        onError={handleLoadError}
        title={moduleKey}
        sandbox="allow-scripts allow-same-origin allow-forms allow-popups"
      />
    </div>
  )
}
```

### 6.5 Socket.IO 实时数据转发

iAGS 旧模块依赖 Socket.IO 接收实时数据（device、control、warning 等）。在 iframe 模式下，由 Shell 统一管理 Socket.IO 连接，通过 postMessage 转发给 iframe：

```typescript
// src/services/realtimeBridge.ts

import { io, Socket } from 'socket.io-client'

class RealtimeBridge {
  private socket: Socket | null = null
  private iframeRefs: Map<string, HTMLIFrameElement> = new Map()

  connect() {
    this.socket = io('/', {
      path: '/socket.io',
      transports: ['websocket', 'polling'],
    })

    this.socket.on('connect', () => {
      console.log('[RealtimeBridge] Socket.IO connected')
    })

    // 监听所有实时数据事件，转发到对应的 iframe
    const realtimeEvents = [
      'RealTimeData',
      'warningData',
      'device_operation_record',
      'control_state',
      'run_settings_result',
    ]

    realtimeEvents.forEach(eventName => {
      this.socket?.on(eventName, (data) => {
        this.broadcastToAllIframes(eventName, data)
      })
    })
  }

  private broadcastToAllIframes(event: string, data: any) {
    this.iframeRefs.forEach((iframe) => {
      iframe.contentWindow?.postMessage({
        source: 'iags-shell',
        type: 'REALTIME_DATA',
        payload: { event, data },
        timestamp: Date.now(),
        messageId: crypto.randomUUID(),
      }, window.location.origin)
    })
  }

  registerIframe(key: string, iframe: HTMLIFrameElement) {
    this.iframeRefs.set(key, iframe)
  }

  unregisterIframe(key: string) {
    this.iframeRefs.delete(key)
  }

  disconnect() {
    this.socket?.disconnect()
  }
}

export const realtimeBridge = new RealtimeBridge()
```

- Cookie 名称: `yujiang-iAGS-x-access-token` (product.id = iAGS)
- Cookie 类型: httpOnly（JS无法读取，XSS防护）
- 免登录 URL: `/app/account/showLogin`, `/app/account/login`, `/app/system/broker/*`, `/biz/overView/getIDC*`

**关键实现点**:

1. **登录页**: React 组件渲染登录表单，POST 到 `/app/account/login`
2. **Cookie 携带**: 所有 broker API 调用使用 `credentials: 'include'`
3. **Token 刷新**: 拦截 401 响应，调用 `/app/account/confirmLogin` 刷新
4. **登出**: POST `/app/account/logout`，清除 Cookie

```typescript
// src/stores/useAuthStore.ts — 认证 Store

import { create } from 'zustand'
import { brokerGet, brokerPost } from '@/services/brokerClient'

interface User {
  userOID: string
  userAID: string
  userName: string
  orgOID: string
  orgName: string
  roles: string[]
  avatar?: string
}

interface ProcessNode {
  processOID: string
  processAID: string
  name: string
  route: string
  icon?: string
  children: ProcessNode[]
  authority: Record<string, number>
  isAuthorized: boolean
}

interface AuthState {
  user: User | null
  processTree: ProcessNode[]
  isAuthenticated: boolean
  isLoading: boolean

  login: (username: string, password: string) => Promise<void>
  logout: () => Promise<void>
  fetchUserInfo: () => Promise<void>
  fetchProcessTree: () => Promise<void>
  checkAuth: () => Promise<boolean>
}

export const useAuthStore = create<AuthState>((set, get) => ({
  user: null,
  processTree: [],
  isAuthenticated: false,
  isLoading: true,

  login: async (username, password) => {
    const res = await fetch('/app/account/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      credentials: 'include',
      body: JSON.stringify({
        loginType: 'userID',
        UserAID: username,
        Password: password,
        isCreateToken: true,
      }),
    })

    if (!res.ok) {
      const err = await res.json()
      throw new Error(err.errmessage || '登录失败')
    }

    // Cookie 已由 iAGS 设置，现在获取用户信息
    await get().fetchUserInfo()
    await get().fetchProcessTree()
    set({ isAuthenticated: true })
  },

  logout: async () => {
    await fetch('/app/account/logout', {
      method: 'POST',
      credentials: 'include',
    })
    set({ user: null, processTree: [], isAuthenticated: false })
  },

  fetchUserInfo: async () => {
    const result = await brokerGet<User>(['account', 'getMyInfo'])
    set({ user: result as any })
  },

  fetchProcessTree: async () => {
    const result = await brokerGet<ProcessNode[]>(['org', 'getUserProcessTree'])
    set({ processTree: (result as any)?.children || [] })
  },

  checkAuth: async () => {
    set({ isLoading: true })
    try {
      const result = await brokerGet<{ valid: boolean }>(['account', 'checkLogin'])
      if ((result as any)?.valid) {
        await get().fetchUserInfo()
        await get().fetchProcessTree()
        set({ isAuthenticated: true, isLoading: false })
        return true
      }
    } catch {
      // Token 无效
    }
    set({ isAuthenticated: false, isLoading: false })
    return false
  },
}))
```

```typescript
function buildMenuFromProcessTree(
  nodes: ProcessNode[],
  menuConfig: MenuConfigItem[]
): MenuItem[] {
  // 将 iAGS process tree 与菜单配置匹配
  // 只显示用户有权限的菜单项
  return menuConfig
    .filter((config) => {
      const node = findProcessNode(nodes, config.processAID)
      return node?.isAuthorized ?? false
    })

      const node = findProcessNode(nodes, config.processAID)
      return {
        key: config.key,
        label: node?.name || config.label,
        icon: config.icon,
        type: config.type,
        children: config.children
          ? buildMenuFromProcessTree(nodes, config.children)
          : undefined,
      }
    })
}
```

```typescript
// src/router/AuthGuard.tsx

export function AuthGuard({ children }: { children: React.ReactNode }) {
  const { isAuthenticated, isLoading, checkAuth } = useAuthStore()
  const location = useLocation()

  useEffect(() => {
    if (!isAuthenticated) {
      checkAuth()
    }
  }, [])

  if (isLoading) {
    return <FullPageSpinner />
  }

  if (!isAuthenticated) {
    return <Navigate to="/login" state={{ from: location }} replace />
  }

  return <>{children}</>
}
```
